üîÅ Control Flow

Conditionals

if age > 18:
    print("Adult")
elif age == 18:
    print("Exactly 18")
else:
    print("Minor")

# Ternary operator
status = "Adult" if age >= 18 else "Minor"

Loops

# For loop
for i in range(5):       # 0 to 4
    print(i)

for item in [1, 2, 3]:   # Iterate list
    print(item)

# While loop
while condition:
    # Loop body
    if should_exit:
        break     # Exit loop
    if skip_rest:
        continue  # Next iteration

Loop Techniques

# With index
for i, value in enumerate(['a', 'b', 'c']):
    print(f"Index {i}: {value}")

# Multiple sequences
for a, b in zip([1, 2], ['x', 'y']):
    print(a, b)

# Dictionary iteration
for key, value in my_dict.items():
    print(key, value)

üîß Functions

Function Definition

# Basic function
def greet(name):
    return f"Hello, {name}"

# With type hints (Python 3.5+)
def add(a: int, b: int) -> int:
    return a + b

# Default arguments
def power(base, exponent=2):
    return base ** exponent

# Variable arguments
def sum_all(*args):
    return sum(args)

# Keyword arguments
def create_user(**kwargs):
    print(kwargs)  # Becomes dictionary

# Combined
def complex_func(a, b=1, *args, **kwargs):
    print(a, b, args, kwargs)

Lambda Functions

# Anonymous functions
square = lambda x: x ** 2
sorted(names, key=lambda x: x.lower())
Pro Tip: Use *args for variable positional arguments and **kwargs for variable keyword arguments.

üéí Data Structures

Lists

# Creation
nums = [1, 2, 3]
mixed = [1, "a", True]

# Access
nums[0]       # 1 (first item)
nums[-1]      # 3 (last item)
nums[1:3]     # [2, 3] (slice)
nums[::2]     # [1, 3] (step)

# Methods
nums.append(4)       # Add to end
nums.insert(1, 1.5)  # Insert at index
nums.remove(2)       # Remove by value
nums.pop(0)          # Remove by index
nums.sort()          # In-place sort
sorted_nums = sorted(nums)  # New sorted list

Tuples

# Immutable sequences
coords = (4, 5)
single = (1,)  # Note comma

# Unpacking
x, y = coords
a, *rest = [1, 2, 3]  # a=1, rest=[2,3]

Dictionaries

# Creation
person = {"name": "Alice", "age": 30}
person = dict(name="Alice", age=30)

# Access
person["name"]          # "Alice"
person.get("age", 0)    # 30 (0 if key missing)

# Methods
person.keys()    # ['name', 'age']
person.values()  # ['Alice', 30]
person.items()   # [('name', 'Alice'), ('age', 30)]
person.update({"city": "NY"})  # Merge dictionaries

Sets

# Unique elements
unique = {1, 2, 2, 3}  # {1, 2, 3}

# Operations
a = {1, 2, 3}
b = {3, 4, 5}
a | b  # Union {1, 2, 3, 4, 5}
a & b  # Intersection {3}
a - b  # Difference {1, 2}

Collections Module

from collections import defaultdict, Counter, namedtuple

# Default dict
dd = defaultdict(list)  # Auto-initializes with list()

# Counter
counts = Counter("hello")  # {'h':1, 'e':1, 'l':2, 'o':1}

# Named tuple
Point = namedtuple("Point", ["x", "y"])

Reading Files

    content = f.read()

with open("file.txt") as f:

# Write multiple lines
lines = ["First line", "Second line"]
with open("file.txt", "w") as f:
    f.writelines(lines)

üßµ Error Handling

Try/Except Blocks

try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")
except (TypeError, ValueError) as e:
    print(f"Type or value error: {e}")
except Exception as e:  # Catch-all (use sparingly)
    print(f"Unexpected error: {e}")
else:
    print("Runs only if no exceptions")
finally:
    print("Always runs, for cleanup")

Raising Exceptions

def validate_age(age):
    if age < 0:
        raise ValueError("Age cannot be negative")
    return age

# Custom exceptions
class MyError(Exception):
    pass

Common Exception Types

Exception When It Occurs
IndexError Sequence index out of range
KeyError Dictionary key not found
TypeError Operation on wrong type
ValueError Correct type but invalid value
FileNotFoundError File doesn't exist
ImportError Module import failed
Pro Tip: Be specific with exception handling. Catching Exception can mask bugs.

üé≠ Object-Oriented Programming (OOP)

Class Basics

class Person:
    # Class attribute (shared by all instances)
    species = "Homo sapiens"
    
    # Constructor
    def __init__(self, name: str, age: int):
        # Instance attributes
        self.name = name
        self.age = age
    
    # Instance method
    def greet(self):
        return f"Hi, I'm {self.name}"
    
    # Class method
    @classmethod
    def from_birth_year(cls, name, birth_year):
        age = datetime.now().year - birth_year
        return cls(name, age)
    
    # Static method
    @staticmethod
    def is_adult(age):
        return age >= 18

# Usage
p = Person("Alice", 30)
p.greet()

Inheritance

class Employee(Person):
    def __init__(self, name, age, employee_id):
        super().__init__(name, age)  # Parent init
        self.employee_id = employee_id
    
    # Method override
    def greet(self):
        original = super().greet()
        return f"{original} (ID: {self.employee_id})"

# Multiple inheritance
class Manager(Employee, TeamLeader):
    pass

Special Methods

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    # String representation
    def __str__(self):
        return f"Vector({self.x}, {self.y})"
    
    # Math operations
    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)
    
    # Comparison
    def __eq__(self, other):
        return self.x == other.x and self.y == other.y
    
    # Length
    def __len__(self):
        return 2
    
    # Many more: __getitem__, __setitem__, __iter__, etc.

Properties

class Circle:
    def __init__(self, radius):
        self._radius = radius  # Protected
    
    @property
    def radius(self):
        return self._radius
    
    @radius.setter
    def radius(self, value):
        if value < 0:
            raise ValueError("Radius cannot be negative")
        self._radius = value
    
    @property
    def area(self):
        return math.pi * self._radius ** 2

‚è≥ Comprehensions

List Comprehensions

# Basic
squares = [x**2 for x in range(10)]

# With condition
even_squares = [x**2 for x in range(10) if x % 2 == 0]

# Nested
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
flattened = [num for row in matrix for num in row]

Dictionary Comprehensions

# Basic
squares = {x: x**2 for x in range(5)}

# From two lists
keys = ['a', 'b', 'c']
values = [1, 2, 3]
mapping = {k: v for k, v in zip(keys, values)}

# With condition
even_squares = {x: x**2 for x in range(10) if x % 2 == 0}

Set Comprehensions

# Basic
unique_lengths = {len(x) for x in ['a', 'bb', 'ccc']}

# From list with duplicates
numbers = [1, 2, 2, 3, 4, 4, 4]
unique = {x for x in numbers}

Generator Expressions

# Similar to list comprehension but lazy
squares_gen = (x**2 for x in range(10))

# Memory efficient
sum_of_squares = sum(x**2 for x in range(1000000))
Watch Out: While comprehensions are concise, complex ones can hurt readability. Use wisely.

‚öôÔ∏è Lambda, Map, Filter, Reduce

Lambda Functions

# Anonymous functions
square = lambda x: x ** 2

# Often used as arguments
sorted(names, key=lambda x: x.lower())

Map

# Apply function to each item
numbers = [1, 2, 3]
doubled = list(map(lambda x: x * 2, numbers))

# With named function
def to_upper(s):
    return s.upper()
    
names = ['alice', 'bob']
upper_names = list(map(to_upper, names))

Filter

# Filter items
numbers = [1, 2, 3, 4, 5]
evens = list(filter(lambda x: x % 2 == 0, numbers))

Reduce

from functools import reduce

# Cumulative operation
numbers = [1, 2, 3, 4]
product = reduce(lambda x, y: x * y, numbers)  # 24
Modern Python: List comprehensions and generator expressions often replace map and filter for better readability.

üß™ Testing

unittest (Standard Library)

import unittest

class TestMath(unittest.TestCase):
    def test_add(self):
        self.assertEqual(1 + 1, 2)
    
    def test_divide(self):
        with self.assertRaises(ZeroDivisionError):
            1 / 0

if __name__ == "__main__":
    unittest.main()

pytest (Third-Party)

# test_sample.py
def add(a, b):
    return a + b

def test_add():
    assert add(2, 3) == 5
    assert add(-1, 1) == 0

# Run with: pytest test_sample.py

Mocking

from unittest.mock import Mock, patch

# Create mock
mock = Mock(return_value=42)
mock()  # Returns 42

# Patch in tests
def get_data():
    # Some expensive operation
    return "real data"

@patch("__main__.get_data", return_value="mocked data")
def test_get_data(mock_get):
    assert get_data() == "mocked data"
